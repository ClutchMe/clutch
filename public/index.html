const express = require('express'); const http = require('http'); const socketIO = require('socket.io'); const app = express(); const server = http.createServer(app); const io = socketIO(server); app.use(express.static('public')); io.on('connection', (socket) => { console.log('🟢 A user connected:', socket.id); socket.on('chat message', (msg) => { io.emit('chat message', msg); }); socket.on('disconnect', () => { console.log('🔴 User disconnected:', socket.id); }); }); const PORT = 3000; server.listen(PORT, () => { console.log(`🚀 Clutch server running on http://localhost:${PORT}`); }); Press Ctrl + X, then Y, then Enter to save and exit. 📁 Step 3: Add a Basic Frontend Make the public folder and an index.html: mkdir public nano public/index.html Paste this HTML: Clutch Chat
